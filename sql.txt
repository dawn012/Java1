  `cinema_name` varchar(30) NOT NULL,
  `cinema_address` varchar(80) NOT NULL,
  `cinema_phone` varchar(20) NOT NULL,

CREATE TABLE `hall` (
  `hall_id` int(11) AUTO_INCREMENT NOT NULL,
  `cinema_id` int(11) NOT NULL,
